CREATE TABLE IF NOT EXISTS `BungeeChat_Accounts2` (
 UUID BINARY(16) NOT NULL,
 ChannelType ENUM('GLOBAL','LOCAL','STAFF','HELP','GROUP') NOT NULL,
 Vanished BOOLEAN NOT NULL,
 Messenger BOOLEAN NOT NULL,
 SocialSpy BOOLEAN NOT NULL,
 MutedUntil TIMESTAMP NOT NULL,
 StoredPrefix TEXT,
 StoredSuffix TEXT,
 PRIMARY KEY (UUID)
) DEFAULT CHARSET=utf8

 User BINARY(16) NOT NULL,

 CONSTRAINT FOREIGN KEY (User) REFERENCES `BungeeChat_Accounts` (UUID),
 CONSTRAINT FOREIGN KEY (Ignores) REFERENCES `BungeeChat_Accounts` (UUID)
) DEFAULT CHARSET=utf8;
